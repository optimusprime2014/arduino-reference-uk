= static


// OVERVIEW SECTION STARTS
[#overview]
--

[float]
=== Опис
Ключове слово `static` використовується для створення змінних, які видимі лише одній функції. Однак на відміну від локальних змінних, які створюються і знищуються щоразу під час виклику функції, статичні змінні існують поза межами виклику функції, зберігаючи свої дані між викликами функцій.

Змінні, оголошені як статичні, створюються та ініціалізуються лише під час першого виклику функції.

[source,arduino]

#define randomWalkLowRange -20
#define randomWalkHighRange 20
int stepsize;

int thisTime;

void setup() {
  Serial.begin(9600);
}

void loop() {
  stepsize = 5;
  thisTime = randomWalk(stepsize);
  Serial.println(thisTime);
  delay(10);
}

int randomWalk(int moveSize) {
  static int place; // змінна оголошена як статична
  // значення зберігається між викликами функції без можливості її зміни ззовні

  place = place + (random(-moveSize, moveSize + 1));

  if (place < randomWalkLowRange) {                               // перевірка верхньої межі
    place = randomWalkLowRange + (randomWalkLowRange - place);    // відбиття від межі у позитивному напрямку
  }
  else if (place > randomWalkHighRange) {                         // перевірка нижньої межі
    place = randomWalkHighRange - (place - randomWalkHighRange);  // відбиття від межі у негативному напрямку
  }

  return place;
}
----
